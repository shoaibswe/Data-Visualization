##Creating custom color on features

colors = {'Iris-setosa':'r','Iris-versicolor':'y','Iris-virginica':'b'}
fig, x = plt.subplots()

for i in range(len(iris['SepalLengthCm'])):
    x.scatter(iris['SepalLengthCm'][i],\
    iris['SepalWidthCm'][i],\
    color = colors[iris['Species'][i]])
    
    
    x.set_title('Visualizing Iris data')
    x.set_xlabel('SepalLengthCm')
    x.set_ylabel('SepalWidthCm')
